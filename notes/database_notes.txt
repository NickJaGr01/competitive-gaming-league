SERVER_TABLE:
server_id BIGINT
owner_id BIGINT
requestable_roles JSON DEFAULT '{}' {'rolename': roleid}
default_elo INTEGER DEFAULT 1300
elo_k_factor INTEGER DEFAULT 64
force_usernames BOOLEAN DEFAULT FALSE
team_roles_enabled BOOLEAN DEFAULT FALSE
default_role BIGINT
teams JSON DEFAULT '{}' {'teamid': roleid(or none)}
region_roles_enabled BOOLEAN DEFAULT FALSE
region_roles JSON DEFAULT '{}' {'region': roleid}
games TEXT[] DEFAULT '{"csgo"}'
hoist_roles BOOLEAN DEFAULT FALSE
mention_roles BOOLEAN DEFAULT TRUE

PLAYER_TABLE:
discord_id BIGINT
username TEXT
elo JSON DEFAULT '{}' {'serverid': {'game': elo}}
#steam_id TEXT
teams JSON DEFAULT '{}' {'game': teamid}
awards JSON DEFAULT '{}' {'serverid': ["award"]}
region TEXT
server_roles JSON DEFAULT '{}' {'serverid': [roleid]}

TEAM_TABLE:
team_id BIGINT
team_name TEXT
team_elo JSON DEFAULT '{}' {serverid: teamelo}
game TEXT
primary_players BIGINT[] DEFAULT '{}'
substitute_players BIGINT[] DEFAULT '{}'
owner_id BIGINT
